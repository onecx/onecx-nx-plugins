# tag::namespace[]
  _<namespaceOfTheGenerator>_::
    The base namespace of the project where the application is part of. +
    For the OneCX, use `+@onecx+`.
# end::namespace[]
# tag::flavor[]
  _<flavor>_::           `+ngrx+` (or `+angular+`).
# end::flavor[]
# tag::nameOfTheProduct[]
  _<nameOfTheProduct>_::
    The technical name of the application (product) for a ceartain business case. +
    E.g. `+book-store+` for an application named "Book Store".
# end::nameOfTheProduct[]
# tag::feature[]
  _<feature>_::         
    The name of the feature module to be created. +
    E.g. `+customer-management+` for a feature module named "Customer Management".
# end::feature[]
